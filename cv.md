var numberOfSteps = function(num) {
        function isOdd(n){
            return n % 2 !== 0;
        }
        let steps = 0;
        while (num > 0){
          if (isOdd(num)){
              num -= 1;
              steps += 1;
          }else {
              num /= 2;
              steps += 1;
          }
        }
        return steps;
      };
